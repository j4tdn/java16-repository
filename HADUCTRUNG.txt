SELECT Class.teacher,COUNT(Student.id) as "number of female student" 
FROM Student 
JOIN Class ON Student.classid = Class.id
WHERE Student.gender = "female" AND Student.classid = (SELECT ID FROM Class WHERE Class.name = "12B")
GROUP BY Class.ID;
